GOAL:
Enable real OTP verification without paid SMS APIs (use Firebase for free phone OTP, SendGrid for email OTP)

---

PART 1: FIREBASE MOBILE OTP (FRONTEND)

STEP 1: INSTALL FIREBASE

npm install firebase

---

STEP 2: INITIALIZE FIREBASE

Create firebase.js:

import { initializeApp } from "firebase/app";
import { getAuth } from "firebase/auth";

const firebaseConfig = {
apiKey: "YOUR_KEY",
authDomain: "YOUR_PROJECT.firebaseapp.com",
projectId: "YOUR_PROJECT_ID",
appId: "YOUR_APP_ID"
};

const app = initializeApp(firebaseConfig);
export const auth = getAuth(app);

---

STEP 3: SETUP RECAPTCHA

import { RecaptchaVerifier } from "firebase/auth";

window.recaptchaVerifier = new RecaptchaVerifier(
"recaptcha-container",
{ size: "invisible" },
auth
);

---

STEP 4: SEND OTP

import { signInWithPhoneNumber } from "firebase/auth";

signInWithPhoneNumber(auth, phoneNumber, window.recaptchaVerifier)
.then((confirmationResult) => {
window.confirmationResult = confirmationResult;
console.log("OTP sent");
});

---

STEP 5: VERIFY OTP

window.confirmationResult.confirm(code)
.then((result) => {
const user = result.user;
console.log("Phone verified:", user.phoneNumber);
});

---

PART 2: BACKEND VERIFICATION (FASTAPI)

STEP 6: SEND VERIFIED PHONE TO BACKEND

After Firebase success:

fetch("/api/verify-phone", {

headers: { "Content-Type": "application/json" },
body: JSON.stringify({

from sendgrid import SendGridAPIClient
from sendgrid.helpers.mail import Mail

def send_email_otp(email, otp):
message = Mail(
from_email=os.getenv("EMAIL_FROM"),
to_emails=email,
subject="Your OTP Code",
html_content=f"<strong>Your OTP is {otp}</strong>"
)
